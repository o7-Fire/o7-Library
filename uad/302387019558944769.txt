    local densityObject = mass / volume

    local buoyantForce = calculateBuoyantForce(fluidDensity, object.VolumeSubmerged, gravity)

    local gravityForce = mass * gravity

    local velocityChange = calculateVelocity(acceleration, deltaTime)

    object.Velocity = object.Velocity + velocityChange
    object.Position = object.Position + displacement
end
